``kotlin

// interface allow implementation of function specific to the object
//  implement onclick function specific to the UI
// create function that is commonly shared in different class
interface Vehicle {
    fun start()
    fun stop()
}

class Car : Vehicle {
// click implement object
    override fun start() {
        println("Car started")
    }

    override fun stop() {
        println("Car stopped")
    }
}

class Truck : Vehicle {
// click implement object
    override fun start() {
        println("Truck started")
    }

    override fun stop() {
        println("Truck stopped")
    }
}
```kotlin
